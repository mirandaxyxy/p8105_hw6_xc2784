``` r
set.seed(1)
n_boot <- 5000

boot_samples <-
  weather %>%
  modelr::bootstrap(n = n_boot, id = "strap_id")

boot_results <-
  boot_samples %>%
  mutate(
    fit  = map(strap, ~ lm(tmax ~ tmin + prcp, data = .x)),
    gl   = map(fit, glance),
    coef = map(fit, tidy)
  ) %>%
  mutate(
    r_sq = map_dbl(gl, "r.squared"),
    beta_ratio = map_dbl(
      coef,
      ~ {
        tmp <- .x %>% select(term, estimate)
        
        b_tmin <- tmp %>%
          filter(term == "tmin") %>%
          pull(estimate)
        
        b_prcp <- tmp %>%
          filter(term == "prcp") %>%
          pull(estimate)
        
        b_tmin / b_prcp
      }
    )
  ) %>%
  select(strap_id, r_sq, beta_ratio)
```
